5. Nginx Reverse Proxy & Load Balancing

The Ultimate Web Server & Gateway

# /etc/nginx/sites-available/production-app
upstream backend_cluster {
    least_conn;
    server backend1:3000 weight=3;
    server backend2:3000 weight=2;
    server backend3:3000 backup;
    keepalive 32;
}

server {
    listen 443 ssl http2;
    server_name api.novhun.dev;

    # SSL Configuration
    ssl_certificate /etc/letsencrypt/live/api.novhun.dev/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/api.novhun.dev/privkey.pem;
    
    # Security headers
    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;
    add_header X-Frame-Options "SAMEORIGIN" always;
    
    location /api/ {
        proxy_pass http://backend_cluster/;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }
}

Enable Site

sudo ln -s /etc/nginx/sites-available/app /etc/nginx/sites-enabled/
sudo nginx -t
sudo systemctl reload nginx
